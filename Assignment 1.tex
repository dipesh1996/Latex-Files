\subsection{Equations}
    \begin{equation}
    \sum_{n=0}^{\infty} \frac{af^n}{n!}(x-a)^n
    \label{taylor}
    \end{equation}
    \eqref{taylor} is the famous Taylor series. Use \say{(1)}, not \say{Eq. (1)} or \say{equation (1)}, except at the beginning of a sentence: \say{Equation (1) is . . .}
        {\\Taylor series in a text would be {$\sum_{n=0}^{\infty}\frac{af^n}{n!}(x-a)^n$}.}
    \subsection{Algorithms}

        \ENDIF
    \ENDIF 
\end{algorithmic}
